Текущие данные будут потеряны.')) { saveData(imported); localStorage.setItem('businessSolo_lastModified', Date.now().toString()); updateLastModified(); alert('Данные успешно загружены!'); } } catch (err) { alert('Ошибка: файл повреждён или имеет неправильный формат.'); } }; reader.readAsText(file); fileInput.value = ''; // очищаем, чтобы можно было снова выбрать тот же файл }); // Автоматически обновляем метку времени при каждом сохранении document.addEventListener('data-updated', () => { localStorage.setItem('businessSolo_lastModified', Date.now().toString()); updateLastModified(); }); })();
